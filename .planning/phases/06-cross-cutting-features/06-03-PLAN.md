   namespace Deskbridge.Core.Services;

   public sealed class AppLockState : IAppLockState
   {
       public bool IsLocked { get; private set; }
       public event EventHandler<bool>? LockStateChanged;

       public void Lock()
       {
           if (IsLocked) return;
           IsLocked = true;
           LockStateChanged?.Invoke(this, true);
       }

       public void Unlock()
       {
           if (!IsLocked) return;
           IsLocked = false;
           LockStateChanged?.Invoke(this, false);
       }
   }

   ```csharp
   using Deskbridge.Core.Interfaces;
   using Deskbridge.Core.Models;
   using Wpf.Ui.Controls;

   namespace Deskbridge.Core.Services;

   public sealed class CommandPaletteService : ICommandPaletteService
   {
       private readonly List<CommandEntry> _commands;

       public CommandPaletteService(Func<Task> newConnection, Func<Task> openSettings,
                                     Func<Task> disconnectAll, Func<Task> quickConnect)
       {
           _commands = new List<CommandEntry>
           {
               new("new-connection", "New Connection", null,
                   new[] { "create", "add" }, SymbolRegular.Add24, "Ctrl+N", newConnection),
               new("settings", "Settings", null,
                   new[] { "preferences", "options" }, SymbolRegular.Settings24, null, openSettings),
               new("disconnect-all", "Disconnect All", "Closes every open RDP tab",
                   new[] { "close all", "end all" }, SymbolRegular.PlugDisconnected24, null, disconnectAll),
               new("quick-connect", "Quick Connect", "Connect by typing a hostname",
                   new[] { "qc", "connect to" }, SymbolRegular.PlugConnected24, "Ctrl+T", quickConnect),
           };
       }

       public IEnumerable<CommandEntry> Commands => _commands;

       public int ScoreCommand(CommandEntry command, string query)
       {
           if (string.IsNullOrWhiteSpace(query)) return 0;
           var q = query.Trim();

           // Parity with ConnectionQueryService:
           // Substring: Title=100, Aliases=80 (standing in for Hostname's 80 slot)
           if (command.Title.Contains(q, StringComparison.OrdinalIgnoreCase)) return 100;
           if (command.Aliases.Any(a => a.Contains(q, StringComparison.OrdinalIgnoreCase))) return 80;

           // Subsequence fallback: Title=40
           if (IsSubsequence(q, command.Title)) return 40;
           return 0;
       }

       private static bool IsSubsequence(string query, string target)
       {
           int qi = 0;
           var q = query.ToLowerInvariant();
           var t = target.ToLowerInvariant();
           for (int ti = 0; ti < t.Length && qi < q.Length; ti++)
               if (q[qi] == t[ti]) qi++;
           return qi == q.Length;
       }
   }
   ```

   /// - IsCommand=false: publish ConnectionRequestedEvent for the wrapped ConnectionModel.
   /// - IsCommand=true : invoke the wrapped CommandEntry.ExecuteAsync delegate.
   /// </summary>
   public sealed class CommandPaletteRowViewModel
   {
       public string Title { get; }
       public string? Subtitle { get; }
       public SymbolRegular Icon { get; }
       public string? Shortcut { get; }
       public int Score { get; }
       public bool IsCommand { get; }
       private readonly ConnectionModel? _connection;
       private readonly CommandEntry? _command;
       private readonly IEventBus? _bus;

       public CommandPaletteRowViewModel(ConnectionModel c, int score, IEventBus bus)
       {
           _connection = c;
           _bus = bus;
           Title = c.Name;
           // UI-SPEC §Row anatomy: Subtitle hidden when Hostname equals Name.
           Subtitle = string.Equals(c.Name, c.Hostname, StringComparison.Ordinal) ? null : c.Hostname;
           Icon = SymbolRegular.Desktop24;
           Shortcut = null;
           Score = score;
           IsCommand = false;
       }

       public CommandPaletteRowViewModel(CommandEntry cmd, int score)
       {
           _command = cmd;
           Title = cmd.Title;
           Subtitle = cmd.Subtitle;
           Icon = cmd.Icon;
           Shortcut = cmd.Shortcut;
           Score = score;
           IsCommand = true;
       }

       public Task ExecuteAsync()
       {
           if (IsCommand)
           {
               return _command!.ExecuteAsync();
           }
           else
           {
               _bus!.Publish(new ConnectionRequestedEvent(_connection!));
               return Task.CompletedTask;
           }
       }
   }
   ```

   using CommunityToolkit.Mvvm.Input;
   using Deskbridge.Core.Interfaces;

   namespace Deskbridge.ViewModels;

   public partial class CommandPaletteViewModel : ObservableObject

       private readonly IConnectionQuery _query;

       public CommandPaletteViewModel(IConnectionQuery query, ICommandPaletteService palette, IEventBus bus)
       {

           _palette = palette;
           _bus = bus;
           Refresh();  // Initial empty-state render.
       }

       [ObservableProperty]
       public partial string SearchText { get; set; } = "";

       [ObservableProperty]
       public partial CommandPaletteRowViewModel? SelectedItem { get; set; }

       public ObservableCollection<CommandPaletteRowViewModel> Items { get; } = new();

       partial void OnSearchTextChanged(string value) => Refresh();

       private void Refresh()
       {
           Items.Clear();
           var q = SearchText?.Trim() ?? "";

           if (string.IsNullOrEmpty(q))
           {
               // D-02 empty state: top 5 recents (connections), then the 4 commands.
               foreach (var c in _query.GetRecent(5))
                   Items.Add(new CommandPaletteRowViewModel(c, 0, _bus));
               foreach (var cmd in _palette.Commands)
                   Items.Add(new CommandPaletteRowViewModel(cmd, 0));
           }
           else
           {
               // D-03 ranked merge: IConnectionQuery.Search + ScoreCommand over each CommandEntry.
               // Search already returns ranked — but we re-merge with commands by re-scoring
               // connections symmetrically. We use the Search result's ORDER as the score proxy
               // (first result = highest score) to avoid exposing the scorer's numeric output.
               var searchResults = _query.Search(q);
               var connectionRows = searchResults
                   .Select((c, i) => new CommandPaletteRowViewModel(c, 100 - i, _bus));

               var commandRows = _palette.Commands
                   .Select(cmd => new { Cmd = cmd, Score = _palette.ScoreCommand(cmd, q) })
                   .Where(x => x.Score > 0)
                   .Select(x => new CommandPaletteRowViewModel(x.Cmd, x.Score));

               var merged = commandRows.Cast<CommandPaletteRowViewModel>()
                   .Concat(connectionRows)
                   .OrderByDescending(r => r.Score)
                   .ThenBy(r => r.IsCommand ? 0 : 1)          // Commands first on ties
                   .ThenBy(r => r.Title, StringComparer.OrdinalIgnoreCase);

               foreach (var row in merged) Items.Add(row);
           }

           SelectedItem = Items.FirstOrDefault();
       }

       [RelayCommand]
       public async Task ExecuteSelectedAsync()
       {
           if (SelectedItem is null) return;
           await SelectedItem.ExecuteAsync();
       }
   }
   ```

   **Scoring compromise noted:** IConnectionQuery.Search returns a ranked IReadOnlyList without exposing numeric scores. Rather than refactoring ConnectionQueryService to expose per-result scores (risky change to a Phase 3 surface), we approximate connection scores as `100 - index` so the first Search result gets 100 and later results decay. For CMD-03 parity the important guarantee is that connection results REMAIN in the order IConnectionQuery.Search returned them — the `100 - index` trick preserves that relative order without changing the Phase 3 surface. Commands are scored via ScoreCommand (absolute score 0-100). Final merge by `OrderByDescending(Score).ThenBy(IsCommand).ThenBy(Title)` yields the UI-SPEC ranking.
