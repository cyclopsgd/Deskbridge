Implement the Phase 6 "visible feedback" layer — a custom ItemsControl-based toast stack replacing WPF-UI's single-snackbar presenter (Q1 locked to Option B per user), a 7-event bus subscriber mapping connection lifecycle events to toasts with the exact UI-SPEC copy strings (NOTF-01 / NOTF-03), explicit silence on user-initiated disconnect (NOTF-02), and atomic settings.json persistence for window position + sidebar state (NOTF-04). Depends on Plan 06-01 because ToastSubscriptionService uses IAuditLogger indirectly (via the Serilog redaction we can now trust), and because AppSettings.SecuritySettings block is consumed by Plan 06-04 — building the shared settings surface here once.

Purpose: Pitfall 3 documents that WPF-UI's SnackbarPresenter is a FIFO queue with ONE visible snackbar at a time — incompatible with UI-SPEC D-07's "max 3 visible simultaneously, newest on top, 4th evicts oldest, hover pauses auto-dismiss" contract. The user locked Option B (custom ItemsControl stack). This plan delivers that custom control while retaining WPF-UI's visual language (same brushes, same SymbolIcon, same ControlAppearance enum for severity colour mapping). Window-state persistence is included in this plan (rather than Plan 06-04) because the settings.json schema is SHARED with Plan 06-04's SecuritySettings — building the full schema ONCE here avoids merge churn later.

2. **Create `src/Deskbridge/ViewModels/ToastStackViewModel.cs`:**
   ```csharp
   using System.Collections.ObjectModel;
   using System.Windows.Threading;
   using Wpf.Ui.Controls;

   namespace Deskbridge.ViewModels;

   /// <summary>
   /// Q1 Option B: custom ItemsControl stack replacing WPF-UI's FIFO SnackbarPresenter.
   /// D-07: max 3 visible, newest on top, 4th push evicts oldest, hover pauses all auto-dismiss timers.
   /// </summary>
   public sealed class ToastStackViewModel
   {
       private const int MaxVisible = 3;
       private long _sequence;
       private readonly Dictionary<Guid, DispatcherTimer> _timers = new();
       private bool _paused;

       public ObservableCollection<ToastItemViewModel> Items { get; } = new();

       public ToastItemViewModel Push(
           string title, string message, ControlAppearance appearance,
           SymbolRegular icon, TimeSpan? duration)
       {
           var item = new ToastItemViewModel
           {
               Sequence = Interlocked.Increment(ref _sequence),
               Appearance = appearance,
               Icon = icon,
               Title = title,
               Message = message,
               Duration = duration,
           };
           item.DismissRequested += (_, _) => Remove(item);

           // D-07 eviction BEFORE insert so the 4th item always fits.
           while (Items.Count >= MaxVisible)
           {
               // Oldest = lowest Sequence. Items are maintained with NEWEST AT INDEX 0 so the
               // oldest is at the end. Remove Items[^1].
               Remove(Items[^1]);
           }

           Items.Insert(0, item);  // Newest on top

           if (duration.HasValue)
               StartTimer(item, duration.Value);

           return item;
       }

       private void StartTimer(ToastItemViewModel item, TimeSpan duration)
       {
           var timer = new DispatcherTimer(DispatcherPriority.Normal,
                                            System.Windows.Application.Current?.Dispatcher
                                              ?? Dispatcher.CurrentDispatcher)
           {
               Interval = duration,
           };
           timer.Tick += (_, _) =>
           {
               timer.Stop();
               _timers.Remove(item.Id);
               Remove(item);
           };
           _timers[item.Id] = timer;
           if (!_paused) timer.Start();
       }

       private void Remove(ToastItemViewModel item)
       {
           if (_timers.TryGetValue(item.Id, out var timer))
           {
               timer.Stop();
               _timers.Remove(item.Id);
           }
           Items.Remove(item);
       }

       public void Pause()
       {
           _paused = true;
           foreach (var t in _timers.Values) t.Stop();
           foreach (var i in Items) i.IsPaused = true;
       }

       public void Resume()
       {
           _paused = false;
           foreach (var t in _timers.Values) t.Start();
           foreach (var i in Items) i.IsPaused = false;
       }
   }
   ```

   ```csharp
   using Deskbridge.Core.Interfaces;
   using Deskbridge.Core.Models;
   using Deskbridge.ViewModels;
   using Wpf.Ui.Controls;

   namespace Deskbridge.Services;

   /// <summary>
   /// NOTF-01 / NOTF-03: maps 6 bus events to ToastStackViewModel.Push calls with the
   /// exact UI-SPEC §Toast Copywriting strings. Registered as a singleton and eagerly
   /// resolved in App.OnStartup so subscriptions land before the first event.
   /// NOTF-02 compliance: ZERO IContentDialogService calls.
   /// </summary>
   public sealed class ToastSubscriptionService
   {
       private readonly ToastStackViewModel _stack;
       private readonly HashSet<Guid> _reconnectingIds = new();

       public ToastSubscriptionService(IEventBus bus, ToastStackViewModel stack)
       {
           _stack = stack;
           bus.Subscribe<ConnectionEstablishedEvent>(this, OnConnected);
           bus.Subscribe<ConnectionClosedEvent>(this, OnClosed);
           bus.Subscribe<ReconnectingEvent>(this, OnReconnecting);
           bus.Subscribe<ConnectionFailedEvent>(this, OnFailed);
           bus.Subscribe<UpdateAvailableEvent>(this, OnUpdateAvailable);
           bus.Subscribe<ConnectionImportedEvent>(this, OnImported);
       }

       private void OnConnected(ConnectionEstablishedEvent e)
       {
           var host = e.Connection.Hostname;
           if (_reconnectingIds.Remove(e.Connection.Id))
           {
               // Reconnect-succeeded path: "Reconnected" 3s (UI-SPEC line 392).
               _stack.Push(
                   "Reconnected",
                   $"Reconnection to {host} succeeded.",
                   ControlAppearance.Info,
                   SymbolRegular.Info24,
                   TimeSpan.FromSeconds(3));
               // Note: we do NOT explicitly dismiss the sticky Reconnecting toast; the
               // push-with-cap-3 rule evicts it naturally if space is needed. If strict
               // "dismiss-and-replace" is needed, Plan 06-04 UAT can revisit.
           }
           else
           {
               _stack.Push(
                   "Connected",
                   $"Connected to {host}.",
                   ControlAppearance.Info,
                   SymbolRegular.Info24,
                   TimeSpan.FromSeconds(2));
           }
       }

       private void OnClosed(ConnectionClosedEvent e)
       {
           // UI-SPEC deliberate silence on user-initiated close.
           if (e.Reason == DisconnectReason.UserInitiated) return;
           _stack.Push(
               "Disconnected",
               $"Disconnected from {e.Connection.Hostname}.",
               ControlAppearance.Info,
               SymbolRegular.Info24,
               TimeSpan.FromSeconds(3));
       }

       {

           _stack.Push(

               ControlAppearance.Caution,

       {
           _reconnectingIds.Remove(e.Connection.Id);
           _stack.Push(
               "Connection failed",
               $"Could not connect to {e.Connection.Hostname}. {e.Reason}.",
               ControlAppearance.Danger,
               SymbolRegular.ErrorCircle24,
               duration: null);  // Sticky

       {

               "Update available",
               $"Deskbridge {e.Version} is ready to install.",
               ControlAppearance.Info,
               SymbolRegular.Info24,
               duration: null);  // Sticky

       {

               "Import complete",
               $"Imported {e.Count} connection(s) from {e.Source}.",
               ControlAppearance.Info,
               SymbolRegular.Info24,
               TimeSpan.FromSeconds(3));
       }
   ```

   ```csharp
   using System.IO;
   using System.Text;
   using System.Text.Json;
   using Deskbridge.Core.Interfaces;
   using Deskbridge.Core.Settings;
   using Serilog;

   namespace Deskbridge.Core.Services;

   public sealed class WindowStateService : IWindowStateService
   {
       private readonly string _path;

       public WindowStateService() : this(Path.Combine(
           Environment.GetFolderPath(Environment.SpecialFolder.ApplicationData),
           "Deskbridge", "settings.json"))
       { }

       internal WindowStateService(string path)
       {
           _path = path;
           Directory.CreateDirectory(Path.GetDirectoryName(_path)!);
       }

       public async Task<AppSettings> LoadAsync(CancellationToken cancellationToken = default)
       {
           if (!File.Exists(_path)) return new AppSettings();

           try
           {
               var json = await File.ReadAllTextAsync(_path, Encoding.UTF8, cancellationToken).ConfigureAwait(false);
               var loaded = JsonSerializer.Deserialize(json, AppSettingsContext.Default.AppSettings);
               if (loaded is null)
               {
                   Log.Warning("settings.json deserialised to null — returning defaults");
                   return new AppSettings();
               }
               if (loaded.SchemaVersion != 1)
               {
                   Log.Warning("settings.json has unknown SchemaVersion={Version} — returning defaults", loaded.SchemaVersion);
                   return new AppSettings();
               }
               return loaded;
           }
           catch (Exception ex)
           {
               Log.Warning(ex, "Failed to load settings.json — returning defaults");
               return new AppSettings();
           }
       }

       public async Task SaveAsync(AppSettings settings, CancellationToken cancellationToken = default)
       {
           var json = JsonSerializer.Serialize(settings, AppSettingsContext.Default.AppSettings);
           var tmp = _path + ".tmp";
           // Encoding.UTF8 writes NO BOM by default for WriteAllBytes.
           await File.WriteAllTextAsync(tmp, json, new UTF8Encoding(false), cancellationToken).ConfigureAwait(false);
           File.Move(tmp, _path, overwrite: true);   // atomic on NTFS (JsonConnectionStore precedent)
       }
   }
   ```
