   {
       private readonly ConnectionEditorViewModel _viewModel;

       public ConnectionEditorDialog(

       {
           _viewModel = viewModel;
           DataContext = viewModel;
           InitializeComponent();

           if (button == ContentDialogButton.Primary)

               _viewModel.SetPassword(PasswordBox.Password);
               if (!_viewModel.Validate())

           }
           base.OnButtonClick(button);
       }
   }

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| User input -> Editor ViewModel | Connection/group fields, credential input |
| Editor ViewModel -> ICredentialService | Password passed from PasswordBox to credential store |
| Editor ViewModel -> IConnectionStore | Connection/group model persistence |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-09 | Information Disclosure | ConnectionEditorViewModel | mitigate | Password is NOT stored in ViewModel (per Pitfall 5). Passed via SetPassword() from PasswordBox code-behind, immediately forwarded to ICredentialService, then discarded. Never appears in XAML bindings or observable properties. |
| T-03-10 | Tampering | ConnectionEditorViewModel.Save | mitigate | Validate hostname non-empty, port 1-65535 before save. Use ObservableValidator with DataAnnotations. Prevent dialog close on validation failure. |
| T-03-11 | Spoofing | GroupEditorViewModel | accept | Parent group self-reference prevented by excluding current group + descendants from AvailableParentGroups list. Circular group chains prevented at model level. |
| T-03-12 | Information Disclosure | InheritedFromMessage | accept | Shows which group provides credentials (group name only, no credential values). This is intended UX per CONN-09. |
</threat_model>
