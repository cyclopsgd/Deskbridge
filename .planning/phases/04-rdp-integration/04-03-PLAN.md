       ```csharp
       public async Task<bool> RunAsync(
           ConnectionModel connection,
           Func<ConnectionModel, Task<bool>> reconnect,
           Func<int, TimeSpan, Task> notifyAttempt,
           CancellationToken ct)
       {
           for (int attempt = 1; attempt <= MaxAttempts; attempt++)
           {
               var delay = attempt - 1 < BackoffSchedule.Length
                   ? BackoffSchedule[attempt - 1]
                   : CapDelay;

               await notifyAttempt(attempt, delay);
               try { await _delay(delay, ct); }
               catch (OperationCanceledException) { return false; }

               if (ct.IsCancellationRequested) return false;
               if (await reconnect(connection)) return true;
           }
           return false;  // Cap hit — caller switches to manual overlay
       }
       ```

This satisfies the "STA-safe" requirement from RDP-ACTIVEX-PITFALLS §6 without importing PresentationCore. The consumer (ConnectionCoordinator) MUST call this from the UI thread, which it already does via Dispatcher marshal.

       ```csharp
       private void OnDisconnectedAfterConnect(object? sender, int discReason)
       {
           if (sender is not Deskbridge.Protocols.Rdp.RdpHostControl rdp) return;
           if (!_dispatcher.CheckAccess()) { _dispatcher.InvokeAsync(() => OnDisconnectedAfterConnect(sender, discReason)); return; }

           var model = _active?.model;
           if (model is null) return;
           var category = DisconnectReasonClassifier.Classify(discReason);

           // Dispose current host (D-04)
           try { rdp.Dispose(); } catch (Exception ex) { _logger.LogError("Host dispose during reconnect threw: {Type} HResult={HResult:X8}", ex.GetType().Name, ex.HResult); }
           // Note: OnHostUnmounted handled by the ConnectionClosedEvent path — here we suppress that to keep viewport in "reconnecting" state.

           var handle = new ReconnectOverlayHandle();

           if (!DisconnectReasonClassifier.ShouldAutoRetry(category))
           {
               // D-06: auth/licensing -> manual overlay only

               return;

           handle.CancelRequested += (s, e) => _reconnectCts?.Cancel();

       private async Task RunAutoReconnectAsync(ConnectionModel model, ReconnectOverlayHandle handle, CancellationToken ct)
       {
           _bus.Publish(new ReconnectingEvent(model, 1, TimeSpan.FromSeconds(2)));
           var success = await _reconnectCoordinator.RunAsync(
               model,
               reconnect: async m => {
                   var result = await _connect.ConnectAsync(m);
                   return result.Success;
               },
               notifyAttempt: (attempt, delay) => {
                   handle.UpdateAttempt?.Invoke(attempt, delay);
                   _bus.Publish(new ReconnectingEvent(model, attempt, delay));
                   return Task.CompletedTask;
               },
               ct);

           if (!success && !ct.IsCancellationRequested)
           {
               // Cap hit (D-05) -> manual overlay
               handle.SwitchToManual?.Invoke();
               WireManualHandlers(handle, model);
           }
           else if (!success && ct.IsCancellationRequested)
           {
               handle.Close?.Invoke();
               _bus.Publish(new ConnectionClosedEvent(model, DisconnectReason.UserInitiated));
           }
           // On success: the pipeline's ConnectStage already published ConnectionEstablishedEvent, which triggers OnConnectionEstablished -> new active host + HostMounted event -> MainWindow re-mounts.
       ```
