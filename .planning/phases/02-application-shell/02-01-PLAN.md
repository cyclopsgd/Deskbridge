    Keep the existing Title property. Add a `using Deskbridge.Models;` and `using System.Collections.ObjectModel;` at the top (or rely on GlobalUsings if already present). Add these members:

    ```csharp
    // Panel state (per D-04 VS Code toggle pattern)
    [ObservableProperty]
    [NotifyPropertyChangedFor(nameof(IsPanelVisible))]
    [NotifyPropertyChangedFor(nameof(IsConnectionsActive))]
    [NotifyPropertyChangedFor(nameof(IsSearchActive))]
    [NotifyPropertyChangedFor(nameof(IsSettingsActive))]
    private partial PanelMode ActivePanelMode { get; set; } = PanelMode.None;

    public bool IsPanelVisible => ActivePanelMode != PanelMode.None;
    public bool IsConnectionsActive => ActivePanelMode == PanelMode.Connections;
    public bool IsSearchActive => ActivePanelMode == PanelMode.Search;
    public bool IsSettingsActive => ActivePanelMode == PanelMode.Settings;

    // Tab state
    [ObservableProperty]
    [NotifyPropertyChangedFor(nameof(HasNoTabs))]
    private partial ObservableCollection<TabItemViewModel> Tabs { get; set; } = [];

    [ObservableProperty]
    private partial TabItemViewModel? ActiveTab { get; set; }

    public bool HasNoTabs => Tabs.Count == 0;

    // Status bar
    [ObservableProperty]
    private partial string StatusText { get; set; } = "Ready";

    [ObservableProperty]
    private partial string StatusSecondary { get; set; } = "No active connection";

    // Commands
    [RelayCommand]
    private void TogglePanel(PanelMode mode)
    {
        ActivePanelMode = ActivePanelMode == mode ? PanelMode.None : mode;
    }

    [RelayCommand]
    private void CloseTab(TabItemViewModel tab)
    {
        Tabs.Remove(tab);
        if (ActiveTab == tab)
        {
            ActiveTab = Tabs.Count > 0 ? Tabs[^1] : null;
            if (ActiveTab is not null) ActiveTab.IsActive = true;
        }
        OnPropertyChanged(nameof(HasNoTabs));
    }

    [RelayCommand]
    private void SwitchTab(TabItemViewModel tab)
    {
        if (ActiveTab is not null) ActiveTab.IsActive = false;
        ActiveTab = tab;
        tab.IsActive = true;
    }
    ```

    Note: `ObservableCollection<TabItemViewModel> Tabs { get; set; } = [];` uses collection expression syntax (C# 12+). The `[NotifyPropertyChangedFor(nameof(HasNoTabs))]` on the Tabs property handles when the entire collection is replaced. For individual Add/Remove operations, call `OnPropertyChanged(nameof(HasNoTabs))` explicitly in CloseTab. Also subscribe to `Tabs.CollectionChanged` in a partial method if needed, but for Phase 2 placeholder behavior, explicit calls in CloseTab suffice.
