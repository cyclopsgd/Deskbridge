   ```csharp
   using Deskbridge.Core.Models;   // or Deskbridge.Models if TabState landed in WPF project (Plan 01 decision)

   namespace Deskbridge.ViewModels;

   public partial class TabItemViewModel : ObservableObject
   {
       [ObservableProperty]
       public partial string Title { get; set; } = string.Empty;

       [ObservableProperty]
       public partial bool IsActive { get; set; }

       [ObservableProperty]
       [NotifyPropertyChangedFor(nameof(IsConnecting))]
       [NotifyPropertyChangedFor(nameof(IsReconnecting))]
       [NotifyPropertyChangedFor(nameof(IsError))]
       [NotifyPropertyChangedFor(nameof(TooltipText))]
       public partial TabState State { get; set; } = TabState.Connecting;

       /// &lt;summary&gt;Hostname (from ConnectionModel). Populated by MainWindowViewModel OnTabOpened. Used by TooltipText.&lt;/summary&gt;
       [ObservableProperty]
       [NotifyPropertyChangedFor(nameof(TooltipText))]
       public partial string Hostname { get; set; } = string.Empty;

       /// &lt;summary&gt;Optional resolution override for tooltip. Populated by MainWindowViewModel when a session's resolution is known.&lt;/summary&gt;
       [ObservableProperty]
       [NotifyPropertyChangedFor(nameof(TooltipText))]
       public partial (int Width, int Height)? Resolution { get; set; }

       /// &lt;summary&gt;Current reconnect attempt number (1-20) when State == Reconnecting. Zero/null otherwise.&lt;/summary&gt;
       [ObservableProperty]
       [NotifyPropertyChangedFor(nameof(TooltipText))]
       public partial int ReconnectAttempt { get; set; }

       public Guid ConnectionId { get; init; }

       /// &lt;summary&gt;D-12: exactly one of IsConnecting / IsReconnecting / IsError is ever true at a time. UI-SPEC §Per-Tab State Visual Contract.&lt;/summary&gt;
       public bool IsConnecting =&gt; State == TabState.Connecting;
       public bool IsReconnecting =&gt; State == TabState.Reconnecting;
       public bool IsError =&gt; State == TabState.Error;

       /// &lt;summary&gt;UI-SPEC §Copywriting Contract — tab tooltip per state. U+00B7 middle-dot, U+2026 ellipsis, U+00D7 multiplication, U+2014 em-dash.&lt;/summary&gt;
       public string TooltipText =&gt; State switch
       {
           TabState.Connecting =&gt; $"{Hostname} · Connecting…",
           TabState.Connected =&gt; Resolution is ({ } w, { } h) && w &gt; 0 && h &gt; 0
               ? $"{Hostname} · {w}×{h}"
               : $"{Hostname} · —",
           TabState.Reconnecting =&gt; $"{Hostname} · Reconnecting attempt {ReconnectAttempt}/20",
           TabState.Error =&gt; $"{Hostname} · Connection failed — click tab to reconnect",
           _ =&gt; Hostname,
       };
   }
   ```

   ```csharp
   protected override void OnPreviewKeyDown(KeyEventArgs e)
   {
       var ctrl = (Keyboard.Modifiers & ModifierKeys.Control) == ModifierKeys.Control;
       var shift = (Keyboard.Modifiers & ModifierKeys.Shift) == ModifierKeys.Shift;

       // Ctrl+Tab / Ctrl+Shift+Tab — cycle (TAB-03). Cannot be XAML KeyBinding (see MainWindow.xaml line 17 comment).
       if (ctrl && e.Key == Key.Tab)
       {
           CycleTab(forward: !shift);
           e.Handled = true;
           return;
       }

       // Ctrl+F4 — close tab (D-16 alias for Ctrl+W).
       if (ctrl && !shift && e.Key == Key.F4)
       {
           if (ViewModel.ActiveTab is not null)
               ViewModel.CloseTabCommand.Execute(ViewModel.ActiveTab);
           e.Handled = true;
           return;
       }

       // Ctrl+Shift+T — reopen last closed (D-16).
       if (ctrl && shift && e.Key == Key.T)
       {
           ViewModel.ReopenLastClosedCommand.Execute(null);
           e.Handled = true;
           return;
       }

       // Ctrl+1..Ctrl+9 — jump to tab N. Ctrl+9 is the LAST tab (Chrome/VS Code convention).
       if (ctrl && !shift && e.Key >= Key.D1 && e.Key <= Key.D9)
       {
           var count = ViewModel.Tabs.Count;
           if (count == 0) { e.Handled = true; return; }
           int idx = e.Key == Key.D9 ? count - 1 : (int)(e.Key - Key.D1);
           if (idx >= 0 && idx < count)
               ViewModel.SwitchTabCommand.Execute(ViewModel.Tabs[idx]);
           e.Handled = true;
           return;
       }

       base.OnPreviewKeyDown(e);
   }

   private void CycleTab(bool forward)
   {
       var count = ViewModel.Tabs.Count;
       if (count == 0) return;
       var currentIndex = ViewModel.ActiveTab is null ? -1 : ViewModel.Tabs.IndexOf(ViewModel.ActiveTab);
       var nextIndex = forward
           ? (currentIndex + 1) % count
           : (currentIndex - 1 + count) % count;
       ViewModel.SwitchTabCommand.Execute(ViewModel.Tabs[nextIndex]);
   }

   // Expose ViewModel via a typed property so OnPreviewKeyDown can call commands.
   private MainWindowViewModel ViewModel => (MainWindowViewModel)DataContext;
   ```
